  -- Table: importer.queue_items

-- DROP TABLE IF EXISTS importer.queue_items;

CREATE TABLE IF NOT EXISTS importer.queue_items
(
    id integer NOT NULL DEFAULT nextval('importer.queue_items_id_seq'::regclass),
    source_id integer NOT NULL,
    "position" integer NOT NULL,
    title text COLLATE pg_catalog."default" NOT NULL,
    author text COLLATE pg_catalog."default",
    source_url text COLLATE pg_catalog."default" NOT NULL,
    status text COLLATE pg_catalog."default" NOT NULL,
    planned_title text COLLATE pg_catalog."default",
    planned_author text COLLATE pg_catalog."default",
    planned_category_id text COLLATE pg_catalog."default",
    planned_synopsis text COLLATE pg_catalog."default",
    planned_cover_mode text COLLATE pg_catalog."default" NOT NULL DEFAULT 'source'::text,
    planned_cover_url text COLLATE pg_catalog."default",
    planned_by text COLLATE pg_catalog."default",
    planned_at timestamp with time zone,
    attempts integer NOT NULL DEFAULT 0,
    last_error text COLLATE pg_catalog."default",
    sharebook_book_id uuid,
    metadata_json jsonb,
    created_at timestamp with time zone NOT NULL DEFAULT now(),
    updated_at timestamp with time zone NOT NULL DEFAULT now(),
    retry_count integer DEFAULT 0,
    max_retries integer DEFAULT 3,
    retry_after timestamp with time zone,
    CONSTRAINT queue_items_pkey PRIMARY KEY (id),
    CONSTRAINT queue_items_source_id_position_key UNIQUE (source_id, "position"),
    CONSTRAINT queue_items_source_id_fkey FOREIGN KEY (source_id)
        REFERENCES importer.sources (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT queue_items_planned_cover_mode_check CHECK (planned_cover_mode = 'source'::text),
    CONSTRAINT queue_items_status_check CHECK (status = ANY (ARRAY['waiting_triage'::text, 'triaging'::text, 'triage_rejected'::text, 'waiting_editor'::text, 'editing'::text, 'waiting_process'::text, 'processing'::text, 'done'::text, 'retry_later'::text, 'source_blocked'::text, 'duplicate'::text, 'error'::text]))
)

TABLESPACE pg_default;
